import numpy as np
import sympy as sp
from IPython.display import display, Latex

x = sp.Symbol("x")
f=1

def norm(p1, p2):
    p = sp.expand(p1 * p2)
    return sp.integrate(p, (x, 0, sp.pi))

def p(order, verbose=False):
    vs = [sp.sin((2*i+1)*x) for i in range((order+1)//2)] + [sp.cos((2*i+1)*x) for i in range((order+1)//2)]
    us = [vs[0]/norm(vs[0],vs[0])**0.5]

    # Gram-Schmidt process:
    for v in vs[1:]:
        u_next = v - sum(norm(u, v)/norm(u,u)*u for u in us)
        u_next /= norm(u_next, u_next)**0.5
        us.append(u_next)
    
    if verbose:
        print("Orthogonal polynomials:")
        display(Latex(f"$${sp.latex(us)}$$"))
    
        # Print out the values of (u_i, u_j).
        mat = [[norm(u1, u2) if abs(norm(u1, u2)) > 1e-10 else 0 for u1 in us] for u2 in us]
        mat = np.array([[float('{:.3f}'.format(value)) for value in row] for row in mat])
        print("\n<u_i, u_j> matrix:")
        print(mat)
        print("\n\n\nShould be about 0 everywhere except along the diagonal.")
    
    return sp.expand(sum(u*sp.Integral(f * u, (x, 0, sp.pi)).evalf()/norm(u, u).evalf() for u in us)).evalf()

def yp(t, y):
    # derivative of y
    return 0.5*(t-y)

def exact_y(t, t0=0, y0=1):
    # correct value of y at time t given the initial value problem y(t0)=y0
    return t-2+(y0+2-t0)*np.exp(-0.5*t)/np.exp(-0.5*t0)

def euler(y0, yp, h=0.01, t_start=0, t_end=3):
    ys = [y0]
    y = y0
    t = t_start
    ts = [t]
    local_error = []
    global_error = exact_y
    while t < t_end:
        y += h * yp(t, y)
        ys.append(y)
        t += h
        ts.append(t)
    ts = np.array(ts)
    ys = np.array(ys)
    global_error = (exact_y(ts) - ys)
    local_error = abs(exact_y(ts[1:], ts[:-1], ys[:-1]) - ys[1:])
    return ys[-1], ys, local_error, global_error, ts

for h in [1, 1/2, 1/4, 1/8, 1/100]:
    end, values, local_error, global_error, ts = euler(1, yp, h)
    plt.plot(ts, values, label=f"h={h}")
    print(f"End value for h={h}: {round(end, 5)}",
          f"\tlocal error={round(local_error[-1], 5)}",
