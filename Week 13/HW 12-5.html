Using the Crank-Nicholson method, the first directional problem

$$u_t^{(k+1/2)} = u_{xx}^{(k+1/2)} + u_{yy}^{(k)}$$

becomes

$$\frac{(u_{x,y,t+1}^{(k+1/2)} - u_{x,y, t}^{(k+1/2)})}{\Delta t} = \frac1{2\Delta h^2}\left[u_{x-1,y,t}^{(k+1/2)}-2u_{x,y,t}^{(k+1/2)}+u_{x+1,y,t}^{(k+1/2)} + u_{x-1,y,t+1}^{(k+1/2)}-2u_{x,y,t+1}^{(k+1/2)}+u_{x+1,y,t+1}^{(k+1/2)}\right] + \frac1{2\Delta h^2}\left[u_{x,y-1,t}^{(k)}-2u_{x,y,t}^{(k)}+u_{x,y+1,t}^{(k)} + u_{x,y-1,t+1}^{(k)}-2u_{x,y,t+1}^{(k)}+u_{x,y+1,t+1}^{(k)}\right]$$

Letting $\mu = \frac{\Delta t}{2\Delta h^2}$ and rearranging we have

$$(1+2\mu )u_{x,y,t+1}^{(k+1/2)} - \mu u_{x-1,y,t+1}^{(k+1/2)} - \mu u_{x+1,y,t+1}^{(k+1/2)} = \mu \left[u_{x-1,y,t}^{(k+1/2)}-2u_{x,y,t}^{(k+1/2)}+u_{x+1,y,t}^{(k+1/2)}\right] + \mu \left[u_{x,y-1,t}^{(k)}-2u_{x,y,t}^{(k)}+u_{x,y+1,t}^{(k)} + u_{x,y-1,t+1}^{(k)}-2u_{x,y,t+1}^{(k)}+u_{x,y+1,t+1}^{(k)}\right]$$

and so we can solve for $u_{t+1}^{(k+1/2)}$ with the matrix equation

$$Au_{t+1}^{(k+1/2)} = Bu_t^{(k+1/2)} + c$$

where $A$ is tridiagonal with entries $-\mu, 1+2\mu, -\mu$ as the diagonals and $B$ is also tridiagonal with entries $\mu, -2\mu, \mu$, and $c$ is a vector with the entries for the $u^{(k)}$ terms.

All of this is analogous for the second direction.

Code below (written in Python 3 with numpy and matplotlib).

I'm using $\Delta t = 0.1$ and $n=50$ for this.

import numpy as np

def g(x, y):
    return y*np.sin(np.pi*(x+1))*(1+np.cos(np.pi*y))

n = 50
u = [[g(x/n, y/n) for y in range(1, n)] for x in range(1, n)]
u = np.array(u)

dt = 0.1
mu = dt * n**2 / 2

old_us = [u for i in range(round(1/dt)+1)]

A = np.zeros((n-1, n-1))
for i in range(n-1):
    if i > 0:
        A[i, i-1] = -mu
    if i < n-2:
        A[i, i+1] = -mu
    A[i, i] = 1+2*mu
    
A_inv = np.linalg.inv(A)
    
B = np.zeros((n-1, n-1))
for i in range(n-1):
    if i > 0:
        B[i, i-1] = mu
    if i < n-2:
        B[i, i+1] = mu
    B[i, i] = -2*mu

def x_step(u, t, old_us):
    new_u = np.zeros((n-1, n-1))
    for y in range(n-1):
        c = np.zeros(n-1)
        for x in range(n-1):
            if y > 0:
                c[x] += old_us[t][x, y-1] + old_us[t+1][x, y-1]
            if y < n-2:
                c[x] += old_us[t][x, y+1] + old_us[t+1][x, y+1]
            c[x] -= 2*(old_us[t][x, y] + old_us[t+1][x, y])
        c *= mu
        
        new_u[:, y] = A_inv @ (B@u[:, y] + c)
    return new_u

def crank_nicholson_x(u_start, old_us):
    new_us = [u_start]
    new_u = u_start
    for t in range(round(1/dt)):
        new_u = x_step(new_u, t, old_us)
        new_us.append(new_u)
    
    # for t+1:
    new_us.append(new_u)
    
    return u_start, new_us

def y_step(u, t, old_us):
    new_u = np.zeros((n-1, n-1))
    for x in range(n-1):
        c = np.zeros(n-1)
        for y in range(n-1):
            if x > 0:
                c[y] += old_us[t][x-1, y] + old_us[t+1][x-1, y]
            if x < n-2:
                c[y] += old_us[t][x+1, y] + old_us[t+1][x+1, y]
            c[y] -= 2*(old_us[t][x, y] + old_us[t+1][x, y])
        c *= mu
        
        new_u[x, :] = A_inv @ (B@u[x, :] + c)
    return new_u

def crank_nicholson_y(u_start, old_us):
    new_us = [u_start]
    new_u = u_start
    for t in range(round(1/dt)):
        new_u = y_step(new_u, t, old_us)
        new_us.append(new_u)
    
    # for t+1:
    new_us.append(new_u)
    
    return u_start, new_us

%matplotlib inline
import matplotlib.pyplot as plt
from IPython.display import clear_output

pictures = []

for i in range(1000):
    latest = old_us[-1]
    u, old_us = crank_nicholson_x(u, old_us)
    pictures.append(old_us[-1])
    plt.imshow(old_us[-1], cmap='hot', interpolation='nearest')
    plt.show()
    clear_output(wait=True)
    
    u, old_us = crank_nicholson_y(u, old_us)
    pictures.append(old_us[-1])
    plt.imshow(old_us[-1], cmap='hot', interpolation='nearest')
    plt.show()
    
    if i > 10 and np.amax(latest - old_us[-1]) < 1e-10:
        print(f"Took {i+1} iterations to converge.")
        break
        
    clear_output(wait=True)
